Java Bean Validation Constraints

Here is a list of commonly used Java Bean Validation constraints with examples:

@NotNull

Ensures that the annotated element is not null.

@NotNull
private String name;

@NotEmpty

Ensures that the annotated string is not null and has a length greater than zero.

@NotEmpty
private String email;

@NotBlank

Ensures that the annotated string is not null and has at least one non-whitespace character.

@NotBlank
private String username;

@Size

Specifies the size constraints on a string, collection, or array.

@Size(min = 2, max = 10)
private List<String> tags;
@Size(min = 5, max = 255)
private String description;

@Min

Specifies the minimum value for numeric fields.

@Min(18)
private int age;

@Max

Specifies the maximum value for numeric fields.

@Max(100)
private int score;

@DecimalMin

Specifies the minimum value for decimal numbers.

@DecimalMin("0.5")
private BigDecimal minAmount;

@DecimalMax

Specifies the maximum value for decimal numbers.

@DecimalMax("100.0")
private BigDecimal maxAmount;

@Digits

Specifies the exact number of digits in integral and fractional parts of a number.

@Digits(integer = 3, fraction = 2)
private BigDecimal numericValue;

@Email

Ensures that the annotated string is a valid email address.

@Email
private String emailAddress;

@Pattern

Applies a regular expression pattern constraint.

@Pattern(regexp = "^[A-Za-z0-9]+$")
private String alphanumeric;

@AssertTrue

Ensures that the annotated element is true.

@AssertTrue
private boolean active;

@AssertFalse

Ensures that the annotated element is false.

@AssertFalse
private boolean inactive;

@Future

Ensures that the annotated date is in the future.

@Future
private LocalDate futureDate;

@FutureOrPresent

Ensures that the annotated date is in the future or the present.

@FutureOrPresent
private LocalDateTime futureOrPresentDateTime;

@Past

Ensures that the annotated date is in the past.

@Past
private LocalDate pastDate;

@PastOrPresent

Ensures that the annotated date is in the past or the present.

@PastOrPresent
private LocalDateTime pastOrPresentDateTime;

@Null

Ensures that the annotated element is null.

@Null
private String nullValue;

@Positive

Ensures that the annotated numeric value is positive.

@Positive
private int positiveNumber;

@PositiveOrZero

Ensures that the annotated numeric value is positive or zero.

@PositiveOrZero
private int positiveOrZeroNumber;

@Negative

Ensures that the annotated numeric value is negative.

@Negative
private int negativeNumber;

@NegativeOrZero

Ensures that the annotated numeric value is negative or zero.

@NegativeOrZero
private int negativeOrZeroNumber;

@CreditCardNumber

Validates that a given string is a valid credit card number.

@CreditCardNumber
private String creditCardNumber;

@URL

Ensures that the annotated string is a valid URL.

@URL
private String websiteUrl;
